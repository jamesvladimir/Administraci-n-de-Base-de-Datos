

-- 1. Tablas Base

CREATE TABLE PAIS (
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_PAIS VARCHAR(20) NOT NULL,
    CONSTRAINT PK_PAIS PRIMARY KEY(ID_PAIS)
);

CREATE TABLE CATEGORIA (
    ID_CATEGORIA NUMERIC(5) NOT NULL,
    NOMBRE_CATEGORIA VARCHAR(20) NOT NULL,
    CONSTRAINT PK_CATEGORIA PRIMARY KEY(ID_CATEGORIA)
);

CREATE TABLE EMPLEADOS (
    ID_EMPLEADOS NUMERIC(5) NOT NULL,
    NOMBRE_EMP VARCHAR(20) NOT NULL,
    PATERNO_EMP VARCHAR(20) NOT NULL,
    MATERNO_EMP VARCHAR(20),
    CARGO VARCHAR(20),
    USUARIO VARCHAR(20) NOT NULL,
    CONTRASENA VARCHAR(20) NOT NULL,
    CONSTRAINT PK_EMPLEADOS PRIMARY KEY(ID_EMPLEADOS)
);

-- 2. Cadena de Ubicación 

CREATE TABLE ESTADOS (
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_EDO VARCHAR(20) NOT NULL,
    CONSTRAINT PK_ESTADOS PRIMARY KEY(ID_ESTADOS, ID_PAIS),
    CONSTRAINT FK_EST_PAIS FOREIGN KEY(ID_PAIS) REFERENCES PAIS(ID_PAIS)
);

CREATE TABLE CIUDAD (
    ID_CIUDAD NUMERIC(5) NOT NULL,
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_CIU VARCHAR(20) NOT NULL,
    CONSTRAINT PK_CIUDAD PRIMARY KEY(ID_CIUDAD, ID_ESTADOS, ID_PAIS),
    CONSTRAINT FK_CIU_EST FOREIGN KEY(ID_ESTADOS, ID_PAIS) REFERENCES ESTADOS(ID_ESTADOS, ID_PAIS)
);

CREATE TABLE COLONIA (
    ID_COLONIA NUMERIC(5) NOT NULL,
    ID_CIUDAD NUMERIC(5) NOT NULL,
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_COL VARCHAR(20) NOT NULL,
    CONSTRAINT PK_COLONIA PRIMARY KEY(ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS),
    CONSTRAINT FK_COL_CIU FOREIGN KEY(ID_CIUDAD, ID_ESTADOS, ID_PAIS) REFERENCES CIUDAD(ID_CIUDAD, ID_ESTADOS, ID_PAIS)
);

CREATE TABLE CALLES (
    ID_CALLES NUMERIC(5) NOT NULL,
    ID_COLONIA NUMERIC(5) NOT NULL,
    ID_CIUDAD NUMERIC(5) NOT NULL,
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_CAL VARCHAR(20) NOT NULL,
    CONSTRAINT PK_CALLES PRIMARY KEY(ID_CALLES, ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS),
    CONSTRAINT FK_CAL_COL FOREIGN KEY(ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS) REFERENCES COLONIA(ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS)
);

-- 3. Entidades Principales

CREATE TABLE CLIENTE (
    ID_CLIENTE NUMERIC(8) NOT NULL,
    ID_CALLES NUMERIC(5) NOT NULL,
    ID_COLONIA NUMERIC(5) NOT NULL,
    ID_CIUDAD NUMERIC(5) NOT NULL,
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    NOMBRE_CLIENTE VARCHAR(30) NOT NULL,
    PATERNO_CLI VARCHAR(20) NOT NULL,
    MATERNO_CLI VARCHAR(20),
    TEL_CL NUMERIC(10),
    CORREO VARCHAR(30),
    CONSTRAINT PK_CLIENTE PRIMARY KEY(ID_CLIENTE),
    CONSTRAINT FK_CLI_CALLES FOREIGN KEY(ID_CALLES, ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS) 
        REFERENCES CALLES(ID_CALLES, ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS)
);

CREATE TABLE PROVEEDORES (
    ID_PROVEEDORES NUMERIC(5) NOT NULL,
    ID_CALLES NUMERIC(5) NOT NULL,
    ID_COLONIA NUMERIC(5) NOT NULL,
    ID_CIUDAD NUMERIC(5) NOT NULL,
    ID_ESTADOS NUMERIC(5) NOT NULL,
    ID_PAIS NUMERIC(5) NOT NULL,
    EMPRESA_PROV VARCHAR(50) NOT NULL,
    RFC VARCHAR(20) NOT NULL,
    CORREO VARCHAR(30),
    CONSTRAINT PK_PROVEEDORES PRIMARY KEY(ID_PROVEEDORES),
    CONSTRAINT FK_PROV_CALLES FOREIGN KEY(ID_CALLES, ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS) 
        REFERENCES CALLES(ID_CALLES, ID_COLONIA, ID_CIUDAD, ID_ESTADOS, ID_PAIS)
);

-- 4. Operaciones y Productos

CREATE TABLE PRODUCTO (
    ID_PRODUCTO NUMERIC(8) NOT NULL,
    ID_CATEGORIA NUMERIC(5) NOT NULL,
    ID_PROVEEDORES NUMERIC(5) NOT NULL,
    PRECIO_VENTA DECIMAL(10,2) NOT NULL,
    PRECIO_PROV DECIMAL(10,2) NOT NULL,
    NOMBRE_PRODUCTO VARCHAR(20) NOT NULL,
    CADUCIDAD DATE,
    CONSTRAINT PK_PRODUCTO PRIMARY KEY(ID_PRODUCTO, ID_CATEGORIA),
    CONSTRAINT FK_PROD_CAT FOREIGN KEY(ID_CATEGORIA) REFERENCES CATEGORIA(ID_CATEGORIA),
    CONSTRAINT FK_PROD_PROV FOREIGN KEY(ID_PROVEEDORES) REFERENCES PROVEEDORES(ID_PROVEEDORES)
);

CREATE TABLE INVENTARIO (
    ID_INVENTARIO NUMERIC(8) NOT NULL,
    ID_PRODUCTO NUMERIC(8) NOT NULL,
    ID_CATEGORIA NUMERIC(5) NOT NULL,
    STOCK NUMERIC(5) NOT NULL,
    CONSTRAINT PK_INVENTARIO PRIMARY KEY(ID_INVENTARIO),
    CONSTRAINT FK_INV_PROD FOREIGN KEY(ID_PRODUCTO, ID_CATEGORIA) REFERENCES PRODUCTO(ID_PRODUCTO, ID_CATEGORIA)
);

CREATE TABLE VENTAS (
    ID_VENTAS NUMERIC(10) NOT NULL,
    ID_PRODUCTO NUMERIC(8) NOT NULL,
    ID_CATEGORIA NUMERIC(5) NOT NULL,
    ID_EMPLEADOS NUMERIC(5) NOT NULL,
    ID_CLIENTE NUMERIC(8) NOT NULL,
    CANTIDAD_VENTAS NUMERIC(5) NOT NULL,
    FECHA DATE NOT NULL,
    CONSTRAINT PK_VENTAS PRIMARY KEY(ID_VENTAS),
    CONSTRAINT FK_VTA_PROD FOREIGN KEY(ID_PRODUCTO, ID_CATEGORIA) REFERENCES PRODUCTO(ID_PRODUCTO, ID_CATEGORIA),
    CONSTRAINT FK_VTA_EMP FOREIGN KEY(ID_EMPLEADOS) REFERENCES EMPLEADOS(ID_EMPLEADOS),
    CONSTRAINT FK_VTA_CLI FOREIGN KEY(ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE)
);
